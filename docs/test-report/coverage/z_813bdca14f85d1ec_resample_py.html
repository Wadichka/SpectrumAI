1"""Интерполяция спектра на единую сетку волновых чисел. 

2 

3По умолчанию — 400–4000 см⁻¹ с шагом 1 см⁻¹ (CLAUDE.md §7). Метод — кубический 

4сплайн ``scipy.interpolate.interp1d``. Если исходный диапазон уже целевого, 

5краевые значения заполняются ``out_of_range_fill`` (по умолчанию 0.0) и 

6выводится предупреждение через ``structlog``. 

7""" 

8 

9from __future__ import annotations 

10 

11import numpy as np 

12import numpy.typing as npt 

13import structlog 

14from scipy.interpolate import interp1d 

15 

16from app.preprocessing.config import PreprocessConfig 

17 

18_logger = structlog.get_logger(__name__) 

19 

20 

21def build_target_grid(config: PreprocessConfig) -> npt.NDArray[np.float64]: 

22 """Возвращает фиксированную сетку волновых чисел согласно конфигу.""" 

23 return np.linspace( 

24 config.target_min, 

25 config.target_max, 

26 num=config.target_length, 

27 dtype=np.float64, 

28 ) 

29 

30 

31def resample_to_grid( 

32 wavenumbers: npt.NDArray[np.float64], 

33 intensities: npt.NDArray[np.float64], 

34 *, 

35 config: PreprocessConfig, 

36) -> tuple[npt.NDArray[np.float64], npt.NDArray[np.float64]]: 

37 """Интерполирует спектр на целевую сетку из конфига. 

38 

39 Args: 

40 wavenumbers: исходная ось волновых чисел (см⁻¹), 1D. 

41 intensities: исходные интенсивности, 1D, той же длины. 

42 config: параметры сетки и метода интерполяции. 

43 

44 Returns: 

45 Кортеж ``(target_grid, interpolated_intensities)``, оба длины 

46 ``config.target_length``. 

47 """ 

48 # Часть JCAMP-DX источников выдают убывающие волновые числа — поправим. 

49 if wavenumbers.size > 1 and wavenumbers[0] > wavenumbers[-1]: 

50 order = np.argsort(wavenumbers) 

51 wavenumbers = wavenumbers[order] 

52 intensities = intensities[order] 

53 

54 target_grid = build_target_grid(config) 

55 

56 src_min = float(wavenumbers[0]) 

57 src_max = float(wavenumbers[-1]) 

58 if src_min > config.target_min or src_max < config.target_max: 

59 _logger.warning( 

60 "spectrum_range_does_not_cover_target", 

61 source_min=src_min, 

62 source_max=src_max, 

63 target_min=config.target_min, 

64 target_max=config.target_max, 

65 fill_value=config.out_of_range_fill, 

66 ) 

67 

68 interpolator = interp1d( 

69 wavenumbers, 

70 intensities, 

71 kind=config.interpolation_kind, 

72 bounds_error=False, 

73 fill_value=config.out_of_range_fill, 

74 assume_sorted=True, 

75 ) 

76 resampled = np.asarray(interpolator(target_grid), dtype=np.float64) 

77 return target_grid, resampled
