39 """Принимает list[str], JSON-литерал или строку с запятыми.""" 

40 if isinstance(value, str): 

41 stripped = value.strip() 

42 if stripped.startswith("["): 

43 import json 

44 

45 return json.loads(stripped) 

46 return [item.strip() for item in stripped.split(",") if item.strip()] 

47 return value
